<template>
  <div class="picbook" v-bind:class="[this.orientation]">
    <div class="autoplay" @click="start" v-bind:class="{'active': hideGo}">Go</div>
    <div
      class="start"
      v-bind:class="{'active': pageName === 'start'}"
    >
      <div class="cover">
        <img :src="picbook.coverImageUrl" />
        <div>
          <h2>{{ picbook.name }}</h2>
          <p>Word Count {{ picbook.wordCount }}</p>
          <audio :src="picbook.coverAudioUrl" @ended="startAudioEndedCallback" id="audioPlayer-1" />
        </div>
      </div>
    </div>
    <swiper
      class="swiper"
      ref="mySwiper"
      @slideChangeTransitionStart="changeTransitionStart"
      @slideChangeTransitionEnd="changeTransitionEnd"
      @transitionStart="transitionStart"
      v-bind:class="{'active': pageName === 'swiper'}"
    >
      <swiper-slide
        v-for="(item, index) in pageContents"
        :key="index"
      >
        <img :src="item.imageUrl" />
        <div class="text">
          <p v-if="showTranslation">{{ item.translation }}</p>
          <p v-else>
            <span
              v-for="(textItem, textIndex) in item.text.split(' ')"
              :key="textIndex"
              v-bind:class="{'active': item.readIndex === textIndex}"
            >{{ textItem }}</span>
          </p>
          <span
            class="translation"
            @touchstart="showTranslation = true"
            @touchend="showTranslation = false"
          >译</span>
          <audio
            :id="'audioPlayer'+index"
            :src="item.audioUrl"
            @timeupdate="audioReadCallback"
            @ended="audioReadEndedCallback"
          />
        </div>
      </swiper-slide>
      <div
        class="swiper-pagination"
        slot="pagination"
      ></div>
    </swiper>
    <div
      class="end"
      v-bind:class="{'active': pageName === 'end'}"
    >
      <div class="cover-end">
        <img :src="picbook.coverImageUrl" />
        <div>
          <div @click="restart">Restart o</div>
          <div @click="backToApp">Next ></div>
        </div>
      </div>
    </div>
  </div>
</template>

<script lang="ts">
import { Swiper, SwiperSlide } from "vue-awesome-swiper";
import { Vue, Component } from "vue-property-decorator";
import axios from "axios";

interface PageItem {
  imageUrl: string;
  translation: string;
  text: string;
  readIndex: number;
  [ propName: string ]: any; // 任意类型
}

declare global {
  interface Window {
    yt: any;
  }
}

@Component({
  name: "swiper-example-pagination",
  ['title' as string]: 'Pagination',
  components: {
    Swiper,
    SwiperSlide,
  },
  computed: {
    swiper() {
      return (this.$refs.mySwiper as any).$swiper
    }
  },
})
export default class Home extends Vue {
  title = 'Pagination'
  pageName = '';
  showTranslation = false;
  orientation = ''
  private picbook = {
    name: '',
    coverImageUrl: '',
    coverAudioUrl: '',
    wordCount: 1,
  };
  hideGo = false
  showEnd = false
  isPlaying = false
  playingIndex = 0
  pageContents: PageItem[] = [];
  swiper: any;
  timeupdate: any;
  mounted() {
    // todo
    this.orient();
    window.addEventListener("onorientationchange" in window ? "orientationchange" : "resize", () => {
      this.orient()
    }, false);
  }
  //判断横屏或者竖屏
  orient() {
    if (window.orientation == 0 || window.orientation == 180) {
      this.orientation = 'portrait' // 横向
    } else if (window.orientation == 90 || window.orientation == -90) {
      this.orientation = 'landscape' // 纵向
    }
  }
  asyncData () {
    axios({
      url: 'story.json',
      method: 'get',
    }).then((res: any) => {
      console.log(res);
      this.picbook = res.data.picbook;
      this.pageContents = res.data.picbook.pageContents;
      return 1
    })
    .catch((err) => {
      console.log("抱歉，服务出错！", err);
    });
  }
  async start() {
    this.hideGo = true
    await this.asyncData()
    this.pageName = "start";
    this.$nextTick(async () => {
      await this.delay(1);
      this.audioPlay(-1)
    })
  }
  async restart() {
    this.pageName = 'start'
    await this.delay(0.5)
    this.audioPlay(-1)
  }
  delay(speed: number) {
    return new Promise((resolve) => {
      setTimeout(() => {
        resolve(1)
      }, speed * 1000)
    })
  }
  audioPlay(index: number) {
    this.isPlaying = true
    this.playingIndex = index
    // -1为封面音频 0为page起始音频
    const dom = (document.getElementById(`audioPlayer${index}`) as HTMLAudioElement)
    dom.load()
    dom.play()
    if(index > -1) {
      this.changeTextColor(index)
    }
  }
  resetTextColor(index: number) {
    const item = this.pageContents[index]
    item.readIndex = -1
  }
  changeTextColor(index: number) {
    const item = this.pageContents[index]
    const dom = (document.getElementById(`audioPlayer${index}`) as HTMLAudioElement)
    this.timeupdate = setInterval(() => {
      item.wordStartTimes.forEach((num: number, index: number) => {
        if((dom.currentTime*1000) >= num) {
          item.readIndex = index
        }
      })
    }, 25) as any
  }
  audioPause(index: number) {
    const dom = (document.getElementById(`audioPlayer${index}`) as HTMLAudioElement)
    dom.pause();
    if(this.timeupdate) {
      clearInterval(this.timeupdate)
      this.timeupdate = null
    }
  }
  // 封面音频结束回调
  async startAudioEndedCallback() {
    this.isPlaying = false
    await this.delay(1)
    this.pageName = 'swiper'
    await this.delay(0.5);
    this.audioPlay(0)
  }
  async audioReadEndedCallback() {
    this.isPlaying = false
    // console.log('audioReadEndedCallback', this.swiper.activeIndex)
    const newIndex = this.swiper.activeIndex + 1
    if(this.pageContents.length === newIndex) {
      this.pageName = 'end'
      await this.delay(0.5)
      this.swiper.slideTo(0, 0, false)
    } else { 
      this.swiper.slideTo(newIndex, 500, true)
    }
  }
  audioReadCallback(item: any) {
    //todo
    console.log('audioReadCallback', item)
  }
  changeTransitionStart(item: any) {
    // todo
    console.log('changeTransitionStart', item)
  }
  changeTransitionEnd(item: any) {
    if (this.isPlaying) {
      this.audioPause(this.playingIndex)
      this.resetTextColor(this.playingIndex)
    }
    this.resetTextColor(item.activeIndex)
    // todo
    // console.log('changeTransitionEnd', item)
    this.audioPlay(item.activeIndex)
  }
  transitionStart(item: any) {
    // todo
    console.log('transitionStart', item)
  }
  backToApp() {
    window.yt.backToApp()
  }
}
</script>
